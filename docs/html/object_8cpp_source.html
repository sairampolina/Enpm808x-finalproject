56 bool Object::spawn_pkg() {
57  if (if_spawned) {
58  ROS_INFO_STREAM("[Package Stack]: Package already spawned.");
59  return false;
60  }
61 
62  spawn_pkg_client_ = nh_-> serviceClient
63  <gazebo_msgs::SpawnModel>("/gazebo/spawn_urdf_model");
64 
65  gazebo_msgs::SpawnModel srv;
66  srv.request.model_name = pkg_name_;
67  srv.request.model_xml = urdf_string_;
68 
69  srv.request.initial_pose.position.x = -3.5;
70  srv.request.initial_pose.position.y = -12;
71  srv.request.initial_pose.position.z = 0.025;
72  srv.request.initial_pose.orientation.w = 1;
73  srv.request.reference_frame = "world";
74  srv.response.success = true;
75  spawn_pkg_client_.call(srv);
76 
77 
78  if (srv.response.success) {
79  ROS_INFO_STREAM("[Package Stack]: Package spawned successfully");
80  if_spawned = true;
81 
82  pkg_pose_ = srv.request.initial_pose;
83 
84 
85  pkg_pose_tf_timer_ = nh_->createTimer(ros::Duration(0.1),
86  &Object::publish_pkg_loc, this);
87 
88  update_state_service_ = nh_->advertiseService("/setObjectState",
89  &Object::set_pkg_state_callback, this);
90  } else {
91  ROS_ERROR_STREAM("[Package Stack]: Failed to spawn object");
92  return true;
93  }
94  return false;
95 }
