41 bool wait_for_message(const bool& message_received, double timeout = 5) {
42  ros::Time start = ros::Time::now();
43  ros::Time now;
44  while (!message_received) {
45  ros::spinOnce();
46  now = ros::Time::now();
47  if ((now - start).toSec() > timeout) {
48  return false;
