38 void MyTiago::execution_pipeline() {
39  // Updates the state when it has changed
40  if (pre_state_ != state_) {
41  ROS_INFO_STREAM("Tiago state: " << state_ << "________________");
42  pre_state_ = state_;
43  }
44 
45  // Execution of pick and place pipeleine
46  switch (state_) {
47  case START:
48  state_ = IDLE;
49  break;
50 
51  case IDLE:
52  // Sets waypoints for the robot
53  navigator.set_goal();
54  create_head_client(point_head_client_);
55  state_ = MOVING_TO_WAYPOINT;
56  break;
57 
58  case MOVING_TO_WAYPOINT:
59  // Moves the robot to the assigned waypoint
60  if (detector.if_obj_detected) {
61  navigator.stop_robot();
62  navigator.set_pkgloc_as_goal(get_pkg_pose());
63  state_ = MOVING_TOWARDS_OBJECT;
64  } else if (navigator.if_goal_reached()) {
65  navigator.rot_state_ = Navigation::rotation::ROT_START;
66  state_ = ROTATE_AROUND;
67  }
68  break;
69 
70  case ROTATE_AROUND:
71  // Turns the robot after reaching each waypoint
72  if (detector.if_obj_detected) {
73  navigator.set_pkgloc_as_goal(get_pkg_pose());
74  state_ = MOVING_TOWARDS_OBJECT;
75  } else if (navigator.rot_state_ ==
76  Navigation::rotation::ROT_COMPLETE) {
77  navigator.set_goal();
78  state_ = MOVING_TO_WAYPOINT;
79  } else {
80  navigator.turn_robot();
81  }
82  break;
83 
84  case MOVING_TOWARDS_OBJECT:
85  // Moves the robot towards the object
86  if (if_obj_within_reach()) {
87  navigator.stop_robot();
88  state_ = PICKING_OBJECT;
89  }
90  break;
91 
92  case PICKING_OBJECT:
93  // picks the object
94  pickup_pkg();
95  navigator.set_droploc_as_goal();
96  state_ = MOVING_TO_DROP_LOCATION;
97  break;
98 
99  case MOVING_TO_DROP_LOCATION:
100  // Moving the object to the drop location
101  if (navigator.if_goal_reached())
102  state_ = PLACING_OBJECT;
103  break;
104 
105  case PLACING_OBJECT:
106  // Placing the object
107  place_pkg();
108  ROS_ERROR("[Tiago Stack]: Task accomplished!");
109  state_ = STOP;
110  break;
111  }
112 }

114 bool MyTiago::if_obj_within_reach() {
115  // Checks if object is within reach of the robot arm.
116  auto pkgPose = get_pkg_pose("base_link");
117  double dx = pkgPose.position.x;
118  double dy = pkgPose.position.y;
119  double distance = std::sqrt(dx*dx + dy*dy);
120  ROS_INFO_STREAM(" object is at distance: " << distance);
121  if (distance < 0.7) return true;
122  return false;
123 }

125 geometry_msgs::Pose MyTiago::get_pkg_pose(std::string wrt) {
126  // Chcks if pks is closer to the robot
127  geometry_msgs::Pose pkgPose;
128  if (detector.if_obj_detected == false) {
129  ROS_INFO_STREAM("[Tiago Stack]: object has not been found yet.");
130  } else {
131  geometry_msgs::TransformStamped transformStamped;
132  transformStamped = tfbuffer_.lookupTransform(wrt, "box",
133  ros::Time(0));
134  pkgPose.position.x = transformStamped.transform.translation.x;
135  pkgPose.position.y = transformStamped.transform.translation.y;
136  pkgPose.position.z = transformStamped.transform.translation.z;
137 
138  pkgPose.orientation = transformStamped.transform.rotation;
139  }
140  return pkgPose;
141 }

157 void MyTiago::create_head_client(PointHeadClientPtr& actionClient) {
158  ROS_INFO("[Tiago Stack]: Creating action client to control head.");
159  actionClient.reset(
160  new PointHeadClient("/head_controller/point_head_action"));
161 
162  int iterations = 0, max_iterations = 3;
163  // Wait for head controller action server to come up
164  while (!actionClient->waitForServer(ros::Duration(2.0)) && ros::ok()
165  && iterations < max_iterations) {
166  ROS_DEBUG(R"([Tiago Stack]: Waiting for the point_head_action server to come
167  up)");
168  ++iterations;
169  }
170 
171  if ( iterations == max_iterations )
172  throw std::runtime_error(R"([Tiago Stack]: Error in create_head_client:
173  head controller action server not available)");
174 }
