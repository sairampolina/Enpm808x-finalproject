60 void Manipulation::reach(geometry_msgs::Pose my_pose) {
61  // create am asynchronous spinner
62  ros::AsyncSpinner spinner(1);
63  spinner.start();
64 
65  // publishing robot arm goal pose
66  geometry_msgs::PoseStamped goal;
67  goal.header.frame_id = "base_footprint";
68  goal.pose = my_pose;
69  moveit::planning_interface::MoveGroupInterface group_arm_torso("arm_torso");
70  group_arm_torso.setPlannerId("SBLkConfigDefault");
71  group_arm_torso.setPoseReferenceFrame("base_footprint");
72  group_arm_torso.setPoseTarget(goal);
73 
74  ROS_INFO_STREAM("[Manipulation Stack]: Planning for Manipulation ");
75 
76  group_arm_torso.setStartStateToCurrentState();
77  group_arm_torso.setMaxVelocityScalingFactor(1.0);
78 
79  // Create moveit planner interface.
80  moveit::planning_interface::MoveGroupInterface::Plan my_plan;
81 
82  group_arm_torso.setPlanningTime(5.0);
83  bool flag = static_cast<bool>(group_arm_torso.plan(my_plan));
84 
85  if (!flag)
86  throw std::runtime_error("No plan found");
87 
88  ROS_INFO_STREAM(" Plan found ");
89 
90  // Execute the plan
91  ros::Time start = ros::Time::now();
92 
93  moveit::planning_interface::MoveItErrorCode e = group_arm_torso.move();
94  if (!static_cast<bool>(e))
95  throw std::runtime_error("[Manipulation Stack]: Error in manipulation");
96 
97  spinner.stop();
