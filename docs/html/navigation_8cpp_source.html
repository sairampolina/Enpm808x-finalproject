107 bool Navigation::if_goal_reached() {
108  auto x = pre_pose_.position.x - goal_pose_.position.x;
109  x = std::pow(x, 2);
110  auto y = pre_pose_.position.y - goal_pose_.position.y;
111  y = std::pow(y, 2);
112

114  return false;
115  return true;

118 void Navigation::turn_robot() {
119  if (pose_flag_) {
120  if (rot_state_ == ROT_START) {
121  tf2::fromMsg(pre_pose_.orientation, init_quaternion_);
122  init_quaternion_ = init_quaternion_.inverse();
123 
124  rot_state_ = ROTATING;
125  } else {
126  tf2::Quaternion tf2_quaternion;
127 
128  tf2::fromMsg(pre_pose_.orientation, tf2_quaternion);
129 
130  tf2_quaternion *= init_quaternion_;
131 
132  auto rot_angle = tf2::getYaw(tf2_quaternion);
133 
134  set_rot_vel();
135 
136  if (rot_angle < 0 && rot_angle > -0.1) {
137  rot_state_ = ROT_COMPLETE;
138  }
139  }
140  } else {
141  ROS_INFO_STREAM("[Navigation Stack]: Initial pose not published");
142  }
